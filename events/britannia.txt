	britannia.7 = {
    	type = country_event
    	title = britannia.3.t
    	desc = britannia.3.desc
		picture = throneroom
		hidden = yes

		trigger = {
			OR = {
				AND = {
					exists = scope:northumbria_scope
					exists = scope:wessex_scope
				}
				AND = {
					exists = scope:northumbria_scope
					exists = scope:wales_scope
				}
				AND = {
					exists = scope:wessex_scope
					exists = scope:wales_scope
				}
			}	
		}

    	immediate = {
			trigger_event = {
				id = britannia.7
				days = 1
			}
		}

	    option = {  
			every_province = {
				limit = {
					AND = {
						is_in_region = britain_region
						controller = root
						NOT = {
							owner = root
						}
					}
				}
				set_owned_by = root
			}
			if = {
				limit = { 
					OR = {
						AND = {
							exists = scope:northumbria_scope
							exists = scope:wessex_scope
							NOT = {
								OR = {
									war_with = scope:northumbria_scope
									war_with = scope:wessex_scope
								}
							}
						}
						AND = {
							exists = scope:northumbria_scope
							exists = scope:wales_scope
							NOT = {
								OR = {
									war_with = scope:northumbria_scope
									war_with = scope:wales_scope
								}
							}
						}
						AND = {
							exists = scope:wessex_scope
							exists = scope:wales_scope
							NOT = {
								OR = {
									war_with = scope:wessex_scope
									war_with = scope:wales_scope
								}
							}
						}
					}
				}
				if = {
					limit = {
						exists = scope:northumbria_scope
					}
					scope:northumbria_scope = {
						declare_war_with_wargoal = {
							war_goal = conquer_wargoal
							province = 2003
							target = scope:wessex_scope
						}
						declare_war_with_wargoal = {
							war_goal = conquer_wargoal
							province = 2102
							target = scope:wales_scope
						}
					}
				}
				else = {
					scope:wessex_scope = {
						declare_war_with_wargoal = {
							war_goal = conquer_wargoal
							province = 2102
							target = scope:wales_scope
						}
					}
				}
			}
		}    
	}
